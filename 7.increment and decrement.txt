
#include <stdio.h>

int main()
{
   int a=1;
   printf("a's value is %d\n",a++);
   printf("a's value is %d\n",a);
   
   a=1;
   printf("a's value is %d\n",++a);
   printf("a's value is %d\n",a);
   
   a=1;
   printf("a's value is %d\n",a--);
   printf("a's value is %d\n",a);
   
    
   a=1;
   printf("a's value is %d\n",--a);
   printf("a's value is %d\n",a);
    
    return 0;
}
